Intensity-Level Slicing

Theory

Intensity slicing is used to highlight specific ranges of pixel intensities in an image. It is particularly useful in applications like satellite imagery and medical imaging where certain intensity ranges correspond to meaningful features.

  • Binary Slicing: Pixels within a given range are set to a high value (e.g., white), others are set to low (e.g., black).
  • Range Highlighting: Keeps the original intensities in the chosen range and suppresses intensities outside the range.

Python Code


import cv2
import numpy as np
import matplotlib.pyplot as plt

def intensity_slicing(img, r_min, r_max, high_value=255, low_value=0, binary=True):
    """
    Apply intensity-level slicing to an image.
    """
    sliced_img = np.zeros_like(img, dtype=np.uint8)
    
    if binary:
        # Binary Slicing
        sliced_img[(img >= r_min) & (img <= r_max)] = high_value
    else:
        # Range Highlighting
        sliced_img = np.copy(img)
        sliced_img[(img < r_min) | (img > r_max)] = low_value
    
    return sliced_img

# Load grayscale image
img = cv2.imread('assets/bear.jpg', cv2.IMREAD_GRAYSCALE)

# Apply intensity-level slicing
r_min, r_max = 100, 200
binary_sliced = intensity_slicing(img, r_min, r_max, binary=True)
highlighted = intensity_slicing(img, r_min, r_max, binary=False)

# Display results
plt.figure(figsize=(10, 6))

plt.subplot(2, 2, 1)
plt.imshow(img, cmap='gray')
plt.title("Original Image")
plt.axis('off')

plt.subplot(2, 2, 2)
plt.imshow(binary_sliced, cmap='gray')
plt.title(f"Binary Slicing ({r_min}-{r_max})")
plt.axis('off')

plt.subplot(2, 2, 3)
plt.imshow(highlighted, cmap='gray')
plt.title(f"Range Highlighting ({r_min}-{r_max})")
plt.axis('off')

plt.subplot(2, 2, 4)
plt.hist(img.ravel(), bins=256, range=(0, 256), color='black')
plt.title("Histogram of Original Image")
plt.xlabel("Intensity Value")
plt.ylabel("Frequency")

plt.tight_layout()
plt.show()
